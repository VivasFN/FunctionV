```python #!/usr/bin/env python3
"""
analytics_pipeline.py

A small data analytics pipeline that:
- Loads CSV data
- Cleans and validates records
- Computes summary statistics
- Generates a simple text report
- Caches intermediate results for faster reruns
"""

import csv
import json
import os
import sys
import time
import hashlib
from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Dict, Any, Optional, Iterable, Tuple


# -----------------------------
# Configuration and constants
# -----------------------------

DEFAULT_INPUT_FILE = "data/input_data.csv"
DEFAULT_OUTPUT_DIR = "reports"
CACHE_DIR = ".cache"

REQUIRED_COLUMNS = ["id", "timestamp", "category", "value"]
VALID_CATEGORIES = {"A", "B", "C", "D"}


# -----------------------------
# Data models
# -----------------------------

@dataclass
class Record:
    id: str
    timestamp: datetime
    category: str
    value: float
    raw: Dict[str, Any] = field(default_factory=dict)


@dataclass
class SummaryStats:
    total_records: int
    valid_records: int
    invalid_records: int
    by_category: Dict[str, Dict[str, float]]
    min_value: Optional[float]
    max_value: Optional[float]
    avg_value: Optional[float]


# -----------------------------
# Utility functions
# -----------------------------

def ensure_dir(path: str) -> None:
    if not os.path.isdir(path):
        os.makedirs(path, exist_ok=True)


def hash_file(path: str) -> str:
    sha = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            sha.update(chunk)
    return sha.hexdigest()


def parse_float(value: str) -> Optional[float]:
    try:
        return float(value)
    except (TypeError, ValueError):
        return None


def parse_timestamp(value: str) -> Optional[datetime]:
    # Try a couple of common formats
    for fmt in ("%Y-%m-%d %H:%M:%S", "%Y-%m-%d", "%m/%d/%Y %H:%M"):
        try:
            return datetime.strptime(value, fmt)
        except (TypeError, ValueError):
            continue
    return None


# -----------------------------
# CSV loading and validation
# -----------------------------

def load_csv(path: str) -> Iterable[Dict[str, str]]:
    with open(path, newline="", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        for row in reader:
            yield row


def validate_row(row: Dict[str, str]) -> Tuple[Optional[Record], Optional[str]]:
    missing = [col for col in REQUIRED_COLUMNS if col not in row or row[col] == ""]
    if missing:
        return None, f"Missing required columns: {', '.join(missing)}"

    category = row["category"].strip()
    if category not in VALID_CATEGORIES:
        return None, f"Invalid category: {category}"

    value = parse_float(row["value"])
    if value is None:
        return None, f"Invalid numeric value: {row['value']}"

    ts = parse_timestamp(row["timestamp"])
    if ts is None:
        return None, f"Invalid timestamp: {row['timestamp']}"

    rec = Record(
        id=str(row["id"]).strip(),
        timestamp=ts,
        category=category,
        value=value,
        raw=row,
    )
    return rec, None


def load_and_validate(path: str) -> Tuple[List[Record], List[Dict[str, Any]]]:
    valid: List[Record] = []
    invalid: List[Dict[str, Any]] = []

    for row in load_csv(path):
        rec, error = validate_row(row)
        if rec is not None:
            valid.append(rec)
        else:
            invalid.append({"row": row, "error": error})

    return valid, invalid


# -----------------------------
# Statistics computation
# -----------------------------

def compute_summary(records: List[Record], invalid_count: int) -> SummaryStats:
    total = len(records) + invalid_count
    by_category: Dict[str, Dict[str, float]] = {}
    values: List[float] = []

    for rec in records:
        values.append(rec.value)
        if rec.category not in by_category:
            by_category[rec.category] = {
                "count": 0,
                "sum": 0.0,
                "min": rec.value,
                "max": rec.value,
            }
        cat_stats = by_category[rec.category]
        cat_stats["count"] += 1
        cat_stats["sum"] += rec.value
        cat_stats["min"] = min(cat_stats["min"], rec.value)
        cat_stats["max"] = max(cat_stats["max"], rec.value)

    min_value = min(values) if values else None
    max_value = max(values) if values else None
    avg_value = sum(values) / len(values) if values else None

    # Compute per-category averages
    for cat, stats in by_category.items():
        if stats["count"] > 0:
            stats["avg"] = stats["sum"] / stats["count"]
        else:
            stats["avg"] = None

    return SummaryStats(
        total_records=total,
        valid_records=len(records),
        invalid_records=invalid_count,
        by_category=by_category,
        min_value=min_value,
        max_value=max_value,
        avg_value=avg_value,
    )


# -----------------------------
# Caching layer
# -----------------------------

def cache_key_for_file(path: str) -> str:
    file_hash = hash_file(path)
    return f"{os.path.basename(path)}-{file_hash}.json"


def load_from_cache(path: str) -> Optional[SummaryStats]:
    ensure_dir(CACHE_DIR)
    key = cache_key_for_file(path)
    cache_path = os.path.join(CACHE_DIR, key)
    if not os.path.isfile(cache_path):
        return None

    with open(cache_path, "r", encoding="utf-8") as f:
        data = json.load(f)

    return SummaryStats(
        total_records=data["total_records"],
        valid_records=data["valid_records"],
        invalid_records=data["invalid_records"],
        by_category=data["by_category"],
        min_value=data["min_value"],
        max_value=data["max_value"],
        avg_value=data["avg_value"],
    )


def save_to_cache(path: str, summary: SummaryStats) -> None:
    ensure_dir(CACHE_DIR)
    key = cache_key_for_file(path)
    cache_path = os.path.join(CACHE_DIR, key)

    data = {
        "total_records": summary.total_records,
        "valid_records": summary.valid_records,
        "invalid_records": summary.invalid_records,
        "by_category": summary.by_category,
        "min_value": summary.min_value,
        "max_value": summary.max_value,
        "avg_value": summary.avg_value,
    }

    with open(cache_path, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=2)


# -----------------------------
# Report generation
# -----------------------------

def format_float(value: Optional[float], decimals: int = 2) -> str:
    if value is None:
        return "N/A"
    return f"{value:.{decimals}f}"


def generate_report(summary: SummaryStats, invalid_details: List[Dict[str, Any]]) -> str:
    lines: List[str] = []
    lines.append("DATA ANALYTICS REPORT")
    lines.append("=" * 80)
    lines.append(f"Generated at: {datetime.utcnow().isoformat()} UTC")
    lines.append("")

    lines.append("OVERVIEW")
    lines.append("-" * 80)
    lines.append(f"Total records:   {summary.total_records}")
    lines.append(f"Valid records:   {summary.valid_records}")
    lines.append(f"Invalid records: {summary.invalid_records}")
    lines.append("")

    lines.append("GLOBAL STATISTICS")
    lines.append("-" * 80)
    lines.append(f"Min value: {format_float(summary.min_value)}")
    lines.append(f"Max value: {format_float(summary.max_value)}")
    lines.append(f"Avg value: {format_float(summary.avg_value)}")
    lines.append("")

    lines.append("PER-CATEGORY STATISTICS")
    lines.append("-" * 80)
    if not summary.by_category:
        lines.append("No valid records by category.")
    else:
        for cat, stats in sorted(summary.by_category.items()):
            lines.append(f"Category: {cat}")
            lines.append(f"  Count: {int(stats['count'])}")
            lines.append(f"  Sum:   {format_float(stats['sum'])}")
            lines.append(f"  Min:   {format_float(stats['min'])}")
            lines.append(f"  Max:   {format_float(stats['max'])}")
            lines.append(f"  Avg:   {format_float(stats['avg'])}")
            lines.append("")

    if invalid_details:
        lines.append("INVALID RECORDS (SAMPLE)")
        lines.append("-" * 80)
        sample = invalid_details[:10]
        for idx, item in enumerate(sample, start=1):
            lines.append(f"[{idx}] Error: {item['error']}")
            # Show a compact representation of the row
            compact = {k: v for k, v in item["row"].items()}
            lines.append(f"    Row: {json.dumps(compact, ensure_ascii=False)}")
        if len(invalid_details) > len(sample):
            lines.append(f"... {len(invalid_details) - len(sample)} more invalid records omitted.")
    else:
        lines.append("No invalid records detected.")

    return "\n".join(lines)


def write_report(report: str, output_dir: str) -> str:
    ensure_dir(output_dir)
    timestamp = datetime.utcnow().strftime("%Y%m%d_%H%M%S")
    filename = f"report_{timestamp}.txt"
    path = os.path.join(output_dir, filename)
    with open(path, "w", encoding="utf-8") as f:
        f.write(report)
    return path


# -----------------------------
# Main pipeline
# -----------------------------

def run_pipeline(input_file: str, output_dir: str) -> None:
    start = time.time()
    print(f"[INFO] Starting analytics pipeline")
    print(f"[INFO] Input file: {input_file}")

    cached = load_from_cache(input_file)
    if cached:
        print("[INFO] Using cached summary statistics")
        # We still need invalid details for the report, so recompute them
        _, invalid_details = load_and_validate(input_file)
        report = generate_report(cached, invalid_details)
        report_path = write_report(report, output_dir)
        elapsed = time.time() - start
        print(f"[INFO] Report generated from cache: {report_path}")
        print(f"[INFO] Completed in {elapsed:.2f} seconds")
        return

    print("[INFO] Loading and validating data...")
    valid_records, invalid_details = load_and_validate(input_file)
    print(f"[INFO] Valid records: {len(valid_records)}")
    print(f"[INFO] Invalid records: {len(invalid_details)}")

    print("[INFO] Computing summary statistics...")
    summary = compute_summary(valid_records, len(invalid_details))

    print("[INFO] Saving summary to cache...")
    save_to_cache(input_file, summary)

    print("[INFO] Generating report...")
    report = generate_report(summary, invalid_details)
    report_path = write_report(report, output_dir)

    elapsed = time.time() - start
    print(f"[INFO] Report generated: {report_path}")
    print(f"[INFO] Completed in {elapsed:.2f} seconds")


# -----------------------------
# CLI entry point
# -----------------------------

def parse_args(argv: List[str]) -> Tuple[str, str]:
    input_file = DEFAULT_INPUT_FILE
    output_dir = DEFAULT_OUTPUT_DIR

    if len(argv) >= 2:
        input_file = argv[1]
    if len(argv) >= 3:
        output_dir = argv[2]

    return input_file, output_dir


if __name__ == "__main__":
    inp, out = parse_args(sys.argv)
    if not os.path.isfile(inp):
        print(f"[ERROR] Input file does not exist: {inp}")
        sys.exit(1)

    run_pipeline(inp, out)

    lines.append(f"Avg value: {format_float(summary.avg_value)}")
    lines.append("")

    lines.append("-" * 80)
    if not summary.by_category:
        lines.append("No valid records by category.")
    else:
        for cat, stats in sorted(summary.by_category.items()):
            lines.append(f"Category: {cat}")
            lines.append(f"  Count: {int(stats['count'])}")
            lines.append(f"  Sum:   {format_float(stats['sum'])}")
            lines.append(f"  Min:   {format_float(stats['min'])}")
            lines.append(f"  Max:   {format_float(stats['max'])}")
            lines.append(f"  Avg:   {format_float(stats['avg'])}")
            lines.append("")

    if invalid_details:
        lines.append("INVALID RECORDS (SAMPLE)")
        lines.append("-" * 80)
        sample = invalid_details[:10]
        for idx, item in enumerate(sample, start=1):
            lines.append(f"[{idx}] Error: {item['error']}")

def hash_file(path: str) -> str:
    sha = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            sha.update(chunk)
    return sha.hexdigest()

def parse_timestamp(value: str) -> Optional[datetime]:
    # Try a couple of common formats
    for fmt in ("%Y-%m-%d %H:%M:%S", "%Y-%m-%d", "%m/%d/%Y %H:%M"):
        try:
            return datetime.strptime(value, fmt)
        except (TypeError, ValueError):

    return None

def validate_row(row: Dict[str, str]) -> Tuple[Optional[Record], Optional[str]]:
    missing = [col for col in REQUIRED_COLUMNS if col not in row or row[col] == ""]
    if missing:
        return None, f"Missing required columns: {', '.join(missing)}"

    category = row["category"].strip()
    if category not in VALID_CATEGORIES:
        return None, f"Invalid category: {category}"

    value = parse_float(row["value"])
    if value is None:
        return None, f"Invalid numeric value: {row['value']}"

    ts = parse_timestamp(row["timestamp"])
    if ts is None:
        return None, f"Invalid timestamp: {row['timestamp']}"

    rec = Record(
        id=str(row["id"]).strip(),
        timestamp=ts,
        category=category,
        value=value,
        raw=row,
    )
    return rec, None


def load_and_validate(path: str) -> Tuple[List[Record], List[Dict[str, Any]]]:
    valid: List[Record] = []
    invalid: List[Dict[str, Any]] = []

    for row in load_csv(path):
        rec, error = validate_row(row)
        if rec is not None:
            valid.append(rec)
        else:
            invalid.append({"row": row, "error": error})

    return valid, invalid

def compute_summary(records: List[Record], invalid_count: int) -> SummaryStats:
    total = len(records) + invalid_count
    by_category: Dict[str, Dict[str, float]] = {}
    values: List[float] = []

    for rec in records:
        values.append(rec.value)
        if rec.category not in by_category:
            by_category[rec.category] = {
                "count": 0,
                "sum": 0.0,
                "min": rec.value,
                "max": rec.value,
            }
        cat_stats = by_category[rec.category]
        cat_stats["count"] += 1
        cat_stats["sum"] += rec.value
        cat_stats["min"] = min(cat_stats["min"], rec.value)
        cat_stats["max"] = max(cat_stats["max"], rec.value)

    min_value = min(values) if values else None
    max_value = max(values) if values else None
    avg_value = sum(values) / len(values) if values else None

    # Compute per-category averages
    for cat, stats in by_category.items():
        if stats["count"] > 0:
            stats["avg"] = stats["sum"] / stats["count"]
        else:
            stats["avg"] = None

    return SummaryStats(
        total_records=total,
        valid_records=len(records),
        invalid_records=invalid_count,
        by_category=by_category,
        min_value=min_value,
        max_value=max_value,
        avg_value=avg_value,
    )
